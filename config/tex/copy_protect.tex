% copy_protect.tex
% Automatic copy protection via node processing.
% Randomly wraps text nodes with garbled ActualText for copy/paste protection.
%
% HOW IT WORKS:
% - PDF displays the original readable text
% - When copied, the text is replaced with garbled characters + random words
% - Configurable protection probability (0.0 = no protection, 1.0 = protect all words)
%
% CONFIGURATION:
% Change \protectionProbability to adjust how many words are protected:
%   0.2 = 20% of words (light protection)
%   0.4 = 40% of words (default, balanced)
%   0.7 = 70% of words (aggressive protection)
%   1.0 = 100% of words (maximum protection)
%
\usepackage{luacode}

% Configuration: probability of protecting a word (0.0 to 1.0)
\newcommand{\protectionProbability}{1.0}

\directlua{dofile("config/lua/copy_protect.lua")}
