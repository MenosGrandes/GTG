local function seeded_random(seed)
    local a = 1103515245
    local c = 12345
    local m = 2147483648  -- 2^31
    local state = seed

    return function()
        state = (a * state + c) % m
        return state / m
    end
end

-- Helper: Convert string seed to a number for math.randomseed
local function string_to_seed(str)
    -- Remove .tex extension if present
    local clean_str = str:match("^(.*%.tex)$") and str:gsub("%.tex", "") or str
    local seed = 0
    for i = 1, #clean_str do
        -- Weighted sum of byte values to ensure "ab" != "ba"
        seed = seed + clean_str:byte(i) * i 
    end
	tex.print("\\PackageWarning{LuaInput}{'" .. seed .. "' seed.}")

    return seed
end

-- Fisher-Yates shuffle (MUST match JavaScript)
local function shuffle(array, rng)
    local n = #array
    for i = n, 2, -1 do  -- i from n down to 2
        local j = math.floor(rng() * i) + 1  -- j in [1, i]
        array[i], array[j] = array[j], array[i]
    end
    return array
end

function input_exercise_files(seed, count)
    local dir_name = "exe"
    local count_num = tonumber(count) or 0

    -- 1. Validate Count
    if count_num <= 0 then
        tex.sprint("\\textbf{\\textcolor{red}{ERROR: Count must be a positive number.}}")
        return
    end

    -- 2. Check if folder exists
    local attr = lfs.attributes(dir_name)
    if not attr or attr.mode ~= "directory" then
        tex.sprint("\\textbf{\\textcolor{red}{ERROR: Directory '" .. dir_name .. "' does not exist.}}")
        return
    end

    -- 3. Collect .tex files
    local files = {}
    for file in lfs.dir(dir_name) do
        if file ~= "." and file ~= ".." then
            if file:match("%.tex$") then
                table.insert(files, file)
            end
        end
    end

    -- 4. Check if files exist
    if #files == 0 then
        tex.sprint("\\textbf{\\textcolor{red}{ERROR: No .tex files found in '" .. dir_name .. "'.}}")
        return
    end

    -- 5. Handle Count vs Available
    if count_num > #files then
        tex.sprint("\\textbf{\\textcolor{orange}{Warning: Requested " .. count_num .. " files, but only " .. #files .. " available. Inputting all.}}\\par")
        count_num = #files
    end
    table.sort(files, function(a, b) return a < b end)

    -- 6. Shuffle and Select
    local rng = seeded_random(seed)

    shuffle(files, rng) -- Seed now comes from input or filenames (with .tex stripped)

    -- 7. Input the selected files
    for i = 1, count_num do
        tex.sprint("\\input{" .. dir_name .. "/" .. files[i] .. "}")
		 tex.print("\\PackageWarning{LuaInput}{File '" .. files[i] .. "' added.}")

    end
end
