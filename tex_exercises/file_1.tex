\sectionZadanie

\subsection*{Opis zadania:}
Funkcja odwraca kolejność znaków w podanym ciągu tekstowym.

\subsection*{Nazwa funkcji:}
\func{reverseString}

\subsection*{Parametry wejściowe:}
\begin{itemize}
  \item \texttt{str: string} — ciąg znaków do odwrócenia
\end{itemize}

\subsection*{Zwraca:}
\begin{itemize}
  \item \texttt{string} — ciąg znaków zawierający oryginalny tekst w odwrotnym porządku
\end{itemize}
